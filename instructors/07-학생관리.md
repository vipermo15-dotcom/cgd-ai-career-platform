## 1. CGD 플랫폼 학생 계정 관리

### 1.1 학생 계정 일괄 생성 (학기 초)

```python
# scripts/create_student_accounts.py
import csv
import requests
import os

PLATFORM_API = "https://cgdplatform-pm7dnqip.manus.space/api"
ADMIN_TOKEN = os.getenv("PLATFORM_ADMIN_TOKEN")

def create_accounts_from_csv(csv_file):
    """CSV 파일에서 학생 계정 일괄 생성"""
    with open(csv_file, 'r', encoding='utf-8-sig') as f:
        reader = csv.DictReader(f)
        
        for student in reader:
            response = requests.post(
                f"{PLATFORM_API}/students",
                headers={"Authorization": f"Bearer {ADMIN_TOKEN}"},
                json={
                    "name": student["이름"],
                    "email": student["이메일"],
                    "student_id": student["학번"],
                    "target_job": student.get("희망직군", ""),
                    "send_welcome_email": True
                }
            )
            
            if response.status_code == 201:
                print(f"✓ 계정 생성: {student['이름']}")
            else:
                print(f"✗ 실패: {student['이름']} - {response.text}")

# students_2026.csv 형식:
# 이름,이메일,학번,희망직군
# 홍길동,gildong@student.edu,2026001,그래픽디자이너
```

## 3. 취업률 현황 관리

### 3.1 월별 취업률 집계

```python
# scripts/calculate_employment_rate.py
import json
from datetime import datetime

def calculate_rate(students_data):
    """취업률 계산"""
    total = len(students_data)
    employed = sum(1 for s in students_data if s.get('status') == 'employed')
    job_seeking = sum(1 for s in students_data if s.get('status') == 'seeking')
    not_started = sum(1 for s in students_data if s.get('status') == 'not_started')
    
    rate = (employed / total * 100) if total > 0 else 0
    
    return {
        'date': datetime.now().strftime('%Y-%m'),
        'total': total,
        'employed': employed,
        'job_seeking': job_seeking,
        'not_started': not_started,
        'rate': f"{rate:.1f}%"
    }
```
