## 2. Firecrawl로 기업정보 수집

### 2.1 수집 스크립트

```python
# scripts/build_company_db.py
import os
import json
import time
from datetime import datetime
from firecrawl import FirecrawlApp
from dotenv import load_dotenv

load_dotenv()
app = FirecrawlApp(api_key=os.getenv("FIRECRAWL_API_KEY"))

# 수집할 기업 목록 (초기 설정용)
COMPANIES_TO_COLLECT = [
    # 대기업
    {"name": "삼성전자", "url": "https://careers.samsung.com", "category": "대기업"},
    {"name": "LG전자", "url": "https://careers.lg.com", "category": "대기업"},
    {"name": "카카오", "url": "https://careers.kakao.com", "category": "대기업"},
    {"name": "네이버", "url": "https://recruit.navercorp.com", "category": "대기업"},
    {"name": "현대자동차", "url": "https://careers.hyundai.com", "category": "대기업"},
    
    # 중견기업
    {"name": "한국디자인진흥원", "url": "https://www.kidp.or.kr", "category": "공공기관"},
    {"name": "에이전시한 (가상)", "url": "https://example-agency.kr", "category": "에이전시"},
    
    # 스타트업
    {"name": "토스", "url": "https://toss.im/career", "category": "스타트업"},
    {"name": "당근", "url": "https://team.daangn.com", "category": "스타트업"},
    {"name": "쏘카", "url": "https://www.socar.kr/career", "category": "스타트업"},
]

def collect_company_info(company):
    """단일 기업 정보 수집"""
    print(f"  수집 중: {company['name']}")
    
    try:
        # 기본 정보 스크래핑
        result = app.scrape_url(
            url=company['url'],
            params={
                'formats': ['markdown'],
                'onlyMainContent': True,
                'waitFor': 3000
            }
        )
        
        raw_content = result.get('markdown', '')
        
        # 구조화 데이터 추출 시도
        extracted = app.extract(
            urls=[company['url']],
            params={
                'prompt': """
다음 정보를 추출해주세요:
- 회사 소개 (한 줄)
- 주요 채용 직군 (리스트)
- 인재상 키워드
- 복리후생 주요 항목
- 지원 방법
JSON 형식으로 반환
"""
            }
        )
        
        company_data = {
            **company,
            "raw_content": raw_content[:5000],
            "extracted": extracted,
            "collected_at": datetime.now().isoformat()
        }
        
        return company_data
        
    except Exception as e:
        print(f"  오류: {e}")
        return {**company, "error": str(e), "collected_at": datetime.now().isoformat()}

def build_company_db():
    """전체 기업 DB 구축"""
    print(f"기업 DB 구축 시작: {len(COMPANIES_TO_COLLECT)}개 기업")
    
    all_companies = []
    
    for i, company in enumerate(COMPANIES_TO_COLLECT, 1):
        print(f"\n[{i}/{len(COMPANIES_TO_COLLECT)}] {company['name']}")
        
        data = collect_company_info(company)
        all_companies.append(data)
        
        # 개별 파일 저장
        os.makedirs("data/companies/raw", exist_ok=True)
        date_str = datetime.now().strftime("%Y%m%d")
        with open(f"data/companies/raw/{company['name']}_{date_str}.json", 'w', encoding='utf-8') as f:
            json.dump(data, f, ensure_ascii=False, indent=2)
        
        time.sleep(3)  # 서버 부하 방지
    
    # 마스터 파일 저장
    os.makedirs("data/companies", exist_ok=True)
    with open("data/companies/companies.json", 'w', encoding='utf-8') as f:
        json.dump({
            "last_updated": datetime.now().isoformat(),
            "total_count": len(all_companies),
            "companies": all_companies
        }, f, ensure_ascii=False, indent=2)
    
    print(f"\n완료! data/companies/companies.json 저장")

if __name__ == "__main__":
    build_company_db()
```

### 3.2 CSV → JSON 변환 스크립트

```python
# scripts/csv_to_json.py
import csv
import json

def csv_to_json(csv_file, json_file):
    companies = []
    
    with open(csv_file, 'r', encoding='utf-8-sig') as f:
        reader = csv.DictReader(f)
        for row in reader:
            # design_jobs 필드 처리 (세미콜론 구분)
            if 'design_jobs' in row:
                row['design_jobs'] = row['design_jobs'].split(';')
            
            # boolean 변환
            if 'is_partner' in row:
                row['is_partner'] = row['is_partner'].upper() == 'TRUE'
            
            companies.append(row)
    
    with open(json_file, 'w', encoding='utf-8') as f:
        json.dump({"companies": companies}, f, ensure_ascii=False, indent=2)
    
    print(f"{len(companies)}개 기업 변환 완료")

if __name__ == "__main__":
    csv_to_json("data/companies.csv", "data/companies/companies.json")
```

## 4. 기업 DB 유지 관리

### 4.1 업데이트 주기

| 작업 | 주기 | 담당 |
|------|------|------|
| 채용공고 수집 | 주 2회 (월, 목) | GitHub Actions |
| 기업 기본정보 | 월 1회 | 교수자 |
| 협력 기업 추가 | 수시 | 교수자 |
| 취업 결과 업데이트 | 합격 시 즉시 | 교수자 |

### 4.2 기업 추가 절차

```
1. companies.csv 열기 (Excel)
2. 새 행 추가 (스키마에 맞게)
3. 저장 후 csv_to_json.py 실행
4. data/companies/companies.json 확인
5. git add → commit → push
6. CGD 플랫폼에서 반영 확인 (최대 5분 소요)
```

## 5. 협력 기업 관리

### 5.1 협력 기업 등록 절차

```
1. 기업 담당자 연락 → MOU 또는 구두 협의
2. companies.json에서 is_partner: true 설정
3. CGD 플랫폼 > 협력 기업 섹션에 표시됨
4. 채용 의뢰 수신 시 students/ 학생 목록에 공지
```

### 5.2 협력 기업 혜택 제공 항목

- 졸업생 포트폴리오 열람 권한
- 채용 의뢰 시 학생 추천 서비스
- 기업 홍보물 디자인 작업 연계 (실습 과제)
- 현장 견학 프로그램 우선 참여
