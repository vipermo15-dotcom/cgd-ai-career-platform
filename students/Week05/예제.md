수집 방법: Firecrawl API + Claude 분석

## 예제 2: Firecrawl + Claude 연동 자동화

```python
# 기업 조사 자동화 스크립트
from firecrawl import FirecrawlApp
import anthropic
import os
from dotenv import load_dotenv

load_dotenv()
firecrawl = FirecrawlApp(api_key=os.getenv("FIRECRAWL_API_KEY"))
claude = anthropic.Anthropic(api_key=os.getenv("ANTHROPIC_API_KEY"))

def research_company(company_name, company_url):
    # 1. Firecrawl로 데이터 수집
    scraped = firecrawl.scrape_url(company_url, params={
        'formats': ['markdown'],
        'onlyMainContent': True
    })
    
    raw_content = scraped.get('markdown', '')[:5000]
    
    # 2. Claude로 분석
    message = claude.messages.create(
        model="claude-opus-4-5",
        max_tokens=1500,
        messages=[{
            "role": "user",
            "content": f"""
다음은 {company_name} 채용 페이지 내용입니다:

{raw_content}

그래픽디자인 전공 신입생 관점에서 분석해주세요:
1. 채용 중인 디자인 직군 목록
2. 주요 요구 스킬
3. 인재상 키워드
4. 신입 지원 시 필요한 준비
5. 포트폴리오 제작 포인트
"""
        }]
    )
    
    analysis = message.content[0].text
    
    # 3. 보고서 생성
    report = f"""# {company_name} 기업 조사 보고서

## 원본 데이터 (Firecrawl)
{raw_content[:1000]}...

## Claude 분석 결과
{analysis}
"""
    
    os.makedirs("reports", exist_ok=True)
    with open(f"reports/{company_name}_report.md", 'w', encoding='utf-8') as f:
        f.write(report)
    
    print(f"{company_name} 분석 완료!")
    return report

# 실행
research_company("카카오", "https://careers.kakao.com")
```
